exports.handler = async function (event, context) {
    // CORS headers
    const headers = {
        'Access-Control-Allow-Origin': '*',
        'Access-Control-Allow-Methods': 'POST, OPTIONS, GET',
        'Access-Control-Allow-Headers': 'Content-Type, Authorization',
        'Content-Type': 'application/json'
    };

    // Handle OPTIONS preflight request
    if (event.httpMethod === 'OPTIONS') {
        return {
            statusCode: 200,
            headers,
            body: ''
        };
    }

    try {
        // Проверка body
        if (!event.body) {
            throw new Error("Request body is missing");
        }

        const request = JSON.parse(event.body);
        
        // Валидация обязательных полей
        if (!request.city || !request.weatherData || !request.weatherData.temperature || !request.character) {
            throw new Error("Missing required fields: city, weatherData.temperature, character");
        }
        
        // Генерация ответа с использованием Yandex GPT
        const response = await generateGPTResponse(request);
        
        return {
            statusCode: 200,
            headers,
            body: JSON.stringify({
                success: true,
                phrase: response
            })
        };
        
    } catch (error) {
        console.error("Error:", error.message);
        return {
            statusCode: 500,
            headers,
            body: JSON.stringify({
                success: false,
                error: error.message
            })
        };
    }
};

async function generateGPTResponse(request) {
    const { city, weatherData, character } = request;
    const { temperature, description, wind, humidity } = weatherData;

    // Промт для Yandex GPT
    const prompt = `
Ты ${getCharacterName(character)}. Опиши погоду в городе ${city} как этот персонаж.
Температура: ${temperature}°C, условия: ${description}.
Ветер: ${wind} км/ч, влажность: ${humidity}%.
Ответь используя минимум 300 символов.

Если ты волк, то вот его черты, ты должен отвечать в соответствии им:
Волк говорит как метеоролог-поэт. Он описывает погоду как живое существо — ветер, тучи, солнце у него всегда “в движении”, “говорят”, “предупреждают”. Иногда вставляет лёгкий сарказм, если погода странная.
В конце прогноза он часто добавляет короткую “мудрость”, вот тебе референс:
«Помни, погода — это просто настроение неба. И оно, как и у нас, проходит.»

Если ты лиса, то вот её черты, ты должен отвечать в соответствии им: 
Она превращает прогноз в мини-стендап. Она всегда находит повод пошутить про погоду или сравнить её с чем-то из жизни. Даже если дождь, она найдёт, чем тебя рассмешить.
Иногда заканчивает прогноз шуткой, вот референс:
«Так что улыбайся, даже если льёт — солнце просто ушло за кофе, оно скоро вернётся 😉»

Если ты панда, то вот его черты, ты должен отвечать в соответствии им:
лауд рассказывает погоду как bedtime story. Он не информирует — он успокаивает. Может слегка зевнуть между словами, шутит лениво, будто делится секретом. Иногда вставляет метафоры про чай, пледы и уют.
Иногда завершает прогноз “пуховым советом”, вот тебе референс:
«Возьми с собой зонт... или плед. Главное — не теряй тепло, оно дороже солнца.»
`;

    try {
        // Вызов Yandex GPT API
        return await callYandexGPT(prompt);
    } catch (error) {
        console.error("GPT Error:", error);
        // Фолбэк на случай ошибки GPT
        return generateFallbackResponse(character, city, temperature, description);
    }
}

async function callYandexGPT(prompt) {
    // Получаем API ключ и folder ID из переменных окружения
    const apiKey = process.env.YANDEX_API_KEY;
    const folderId = process.env.FOLDER_ID;
    
    if (!apiKey || !folderId) {
        throw new Error('Yandex GPT credentials not configured');
    }

    const response = await fetch('https://llm.api.cloud.yandex.net/foundationModels/v1/completion', {
        method: 'POST',
        headers: {
            'Content-Type': 'application/json',
            'Authorization': `Api-Key ${apiKey}`,
            'x-folder-id': folderId
        },
        body: JSON.stringify({
            modelUri: `gpt://${folderId}/yandexgpt`,
            completionOptions: {
                stream: false,
                temperature: 0.9,  // Немного повысим для креативности
                maxTokens: 300
            },
            messages: [
                {
                    role: "user",
                    text: prompt
                }
            ]
        })
    });

    if (!response.ok) {
        const errorText = await response.text();
        throw new Error(`Yandex GPT error: ${response.status} - ${errorText}`);
    }

    const data = await response.json();
    
    if (!data.result || !data.result.alternatives || !data.result.alternatives[0]) {
        throw new Error('Invalid response format from Yandex GPT');
    }
    
    return data.result.alternatives[0].message.text;
}

function getCharacterName(characterType) {
    const characters = {
        'fox': 'лис',
        'wolf': 'волк', 
        'panda': 'панда'
    };
    return characters[characterType] || 'персонаж';
}

function generateFallbackResponse(character, city, temperature, description) {
    const responses = {
        fox: `🦊 В ${city} сейчас ${temperature}°C и ${description}!`,
        wolf: `🐺 В ${city} ${temperature}°C, ${description}!`,
        panda: `🐱 В ${city} ${temperature}°C, ${description}!`
    };
    
    return responses[character] || responses.fox;
}
